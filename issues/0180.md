### API Design

```rust
/// Initialize FROST DKG for treasury multisig
pub async fn init_treasury_frost(
    threshold: u16,    // 3 for 3-of-5
    total: u16,         // 5 signers
) -> Result<FrostSession, Error>;

/// Generate commitment for round 1
pub fn generate_commitment(&self) -> Result<Commitment, Error>;

/// Distribute shares for round 2
pub fn distribute_shares(
    &self,
    commitments: Vec<Commitment>,
) -> Result<Vec<Share>, Error>;

/// Verify shares for round 3
pub fn verify_shares(
    &self,
    shares: Vec<Share>,
) -> Result<bool, Error>;

/// Generate signing key package for round 4
pub fn generate_key_package(
    &self,
    verified_shares: Vec<VerifiedShare>,
) -> Result<KeyPackage, Error>;

/// Sign transaction with aggregated signature
pub fn sign_transaction(
    &self,
    key_packages: Vec<KeyPackage>,
    message: &[u8],
) -> Result<AggregatedSignature, Error>;
```
